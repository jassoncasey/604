\frame{\frametitle{Base Environment \& Rules}
   \begin{flushleft}
   \begin{align*}
      \Gamma_0 &::= \{  \mbox{Nat}::\star, \mbox{Bool}::\star, 
                        \mbox{Ascii}::\star, \mbox{Uint x}::\star\} \\
      \Gamma_1 &::= \Gamma_0, \{ 0..\infty:\mbox{Nat}, True:\mbox{Bool}, 
                                 False:\mbox{Bool} \}
   \end{align*}
   \end{flushleft}

\begin{mathpar}
   \inferrule[Kind Basis] {
      \tau :: K \in \Gamma 
   } {
      \Gamma \vdash \tau :: K } \\
   \inferrule[Type Basis] {
      x : \tau \in \Gamma 
   } {
      \Gamma \vdash x : \tau } \\
\end{mathpar}
}

\frame{\frametitle{Abstraction \& Let Rules}
\begin{mathpar}
   \inferrule[Kind Abs]{
      \Sigma | \Gamma \vdash \tau_1 ::\star \\
      \Sigma | \Gamma,x:\tau_1 \vdash \tau_2 ::\star
   } {
      \Gamma \vdash \Pi x:\tau_1.\tau_2 ::\star 
   }\\
   \inferrule[Type Abs]{
      \Sigma | \Gamma \vdash \tau_1 ::\star \\
      \Sigma | \Gamma,x:\tau_1 \vdash t ~:~\tau_2
   } {
      \Sigma | \Gamma \vdash \lambda x:\tau_1.t ~:~ \Pi x:\tau_1.\tau_2
   }\\
   \inferrule[Type Let]{
      \Sigma | \Gamma \vdash t_1 : \tau_1 \\
      \Sigma | \Gamma \vdash \tau_1 ::\star \\
      \Sigma | \Gamma, \langle x:\tau_1 \rangle \vdash t_2 :\tau_2
   } {
      \Sigma | \Gamma \vdash \mbox{let }x = t_1 \mbox{ in } t_2 : \tau_2 
   }\\
   \inferrule[Type TopLevel Let]{
      \Sigma | \Gamma \vdash \tau ::\star \\
      \Sigma | \Gamma \vdash t : \tau \\
      \Sigma | \Gamma, \langle f:\tau \rangle \vdash prgm
   } {
      \Sigma | \Gamma \vdash \mbox{let }f = t:\tau, prgm 
   }
\end{mathpar}
}

\frame{\frametitle{Delta Functions}
   \scriptsize
   \begin{flushleft}
   \begin{align*}
      \mbox{nat\_plus}&: \Pi a:(\mbox{Nat}).\Pi b:(\mbox{Nat}).\mbox{Nat} \\
      \mbox{uint}&: \Pi a:(\mbox{Nat}).\Pi b:(\mbox{Nat}).\mbox{Uint}~a\\
      \mbox{uint\_plus}&: \Pi a:(\Pi x:\mbox{Nat}.\mbox{Uint}~x).
               \Pi b:(\Pi y:\mbox{Nat}.\mbox{Uint}~y).\mbox{Uint}~x>y?x+1:y+1\\
      \mbox{uint\_bits}&: \Pi a:(\Pi x:\mbox{Nat}.\mbox{Uint }x).\mbox{Nat}
   \end{align*}
   \end{flushleft}
}

\frame{\frametitle{Application Rules}
   \begin{mathpar}
   \inferrule[Type App]{
      \Sigma | \Gamma \vdash t_1 : \Pi x:\tau_1.\tau_3 \\
      \Sigma | \Gamma \vdash t_2:\tau_2 \\
      bw \leftarrow BoundVar(\tau_1)~
      bv' \leftarrow Fresh(bv) \\
      \Sigma | \Gamma \vdash [bv'/bv]Body(\tau_1) \equiv \tau_3 \rightsquigarrow \theta
   } {
      \Sigma | \Gamma ~ t_1 ~ t_2 : [t_2/x][bv'/bv]\tau \rightsquigarrow \theta
   }
   \end{mathpar}
}
